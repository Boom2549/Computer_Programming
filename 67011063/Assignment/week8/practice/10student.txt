#include <stdio.h>

int main()
{
    struct name_age {
        char name[50];
        int age;
    };
    struct name_age student[10];
    for(int i = 0; i<10; i++) {
        printf("student[%d]\n", i);
        printf("\tname : ");
        scanf("%s", student[i].name);
        printf("\tage : ");
        scanf("%d", &student[i].age);
    }
    for(int i = 0; i<10; i++) {
        if(student[i].age>20) {
            printf("%s,%d\n", student[i].name, student[i].age);
        }
    }
    return 0;
}
