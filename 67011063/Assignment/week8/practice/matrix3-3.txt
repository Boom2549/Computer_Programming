#include <stdio.h>

int main() {
    int x[3][3];
    int row, column;
    for(row = 0; row<3; row++) {
        for(column = 0; column<3; column++) {
            printf("Enter numbers [%d][%d] : ", row, column);
            scanf("%d", &x[row][column]);
        }
    }
    printf("***Matrix***\n");
    for(int i = 0; i<row; i++) {
        for(int j = 0; j<column; j++) {
            printf("    %d", x[i][j]);
        }
        printf("\n");
    }

    return 0;
}
