#include <stdio.h>

int main() {
    int z[10] = {0, 9, 8, 7, 6, 5, 4, 3, 2, 1};
    int i, t;
    printf("z(Before) = ");
    for(i = 0; i<10; i++) {
        printf("[%d] ", z[i]);
    }
    printf("\n");
    for(int j = 0; j<9; j++) {
        for(int i = 0; i<9-j; i++) {
            if(z[i]>z[i+1]) {
                t = z[i+1];
                z[i+1] = z[i];
                z[i] = t;
            }
        }
    }
    printf("z(After) = ");
    for(int i = 0; i<10; i++) {
        printf("[%d] ", z[i]);
    }
    printf("\n");

    return 0;
}
